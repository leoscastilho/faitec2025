\begin{Glossario}
    % Adicione termos e definições no formato \glossario{TERMO}{DEFINIÇÃO}
    \glossario{Homeostase}{A capacidade do corpo de manter seu ambiente interno em um estado de equilíbrio e estabilidade.}
    \glossario{Psicologia do Esporte}{Um campo de estudo que lida com as emoções, comportamentos e o bem-estar de atletas para otimizar o desempenho.}
    \glossario{Tensão Competitiva}{As emoções que surgem durante os treinos e competições, afetando a performance.}
    \glossario{Teoria dos Quatro Aspectos de Nitsch}{Um modelo para entender como a emoção se relaciona com o desempenho, considerando a situação, o efeito e o resultado.}
    \glossario{Agressividade Hostil}{Comportamento agressivo no esporte que tem a intenção de causar dano a outro.}
    \glossario{Agressividade Instrumental}{Agressividade usada como um meio para alcançar um objetivo competitivo.}
    \glossario{Prática Encoberta}{Técnica em que o atleta treina usando apenas a imaginação.}
    \glossario{Autofala}{O uso da repetição de frases ou palavras para controlar emoções e manter o foco.}
    \glossario{Estabelecimento de Metas}{Planejamento de objetivos claros para melhorar o desempenho.}
    \glossario{Relaxamento}{Técnicas para aliviar tensões e equilibrar a energia do corpo e da mente.}
\end{Glossario}
